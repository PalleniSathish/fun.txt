#include <iostream>
#include <vector>
using namespace std;

int weird_array(vector<int> a) {
  if (a.size() < 2) {
    return 0;
  }

  vector<int> li;
  int count = 0;
  int maximum = 0;

  for (int i = 0; i < a.size(); i++) {
    count = 1;
    for (int j = i + 1; j < a.size(); j++) {
      if (a[i] == a[j]) {
        count++;
      } else {
        li.push_back(count);
        count = 1;
        break;
      }
    }
    li.push_back(count);
  }

  for (int num : li) {
    if (num > maximum) {
      maximum = num;
    }
  }

  return maximum;
}

int main() {
  vector<int> a = {3,3,1,3,3};
  vector<int> b = {5};

  cout << weird_array(a) << endl;
  cout << weird_array(b) << endl;

  return 0;
}
